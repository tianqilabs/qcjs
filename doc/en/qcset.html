QCSet

Description: QCSet is a Set, and the value in the set is unique.

Code

        var qs1 = QCSet();
        console.log(qs1); // QCSet.init {length: 0}

        // String construction;
        var qcs2 = QCSet("a,b,c");
        console.log(qs2); // QCSet.init {0: "a", 1: "b", 2: "c", length: 3}

        // Array creation;
        var qcs3 = QCSet([1, 2, 3]);
        console.log(qs3); // QCSet.init {0: 1, 1: 2, 2: 3, length: 3}

        // Deduplication effect;
        var qcs4 = QCSet('a,b,a,c');
        console.log(qs4); // QCSet.init {0: "a", 1: "b", 2: "c", length: 3}

        // Array deduplication
        var ary = [1,2,3,4,1,4];
        var qcs5 = QCSet(ary);
        console.log(qcs5); // QCSet.init {0: 1, 1: 2, 2: 3, 3: 4, length: 4}

Attributes

QCSet.prototype.length
The number of elements in the QCSet.
Element access
For similar numerical operations, use [index], the index where the index element is located. Returns the element.

Example

        var qcs = QCSet("a, b, c, d");
        
        console.log(qcs.length);    //  4
        console.log(qcs[1]);        // b

Methods

QCSet.prototype.indexOf(arg[, index])
Parameter:

arg: The element to be positioned in the QCSet.

index [Optional]: The index to start the search. If the index is greater than or equal to the length of the QCSet, -1 is returned. If the provided index value is negative, it will be treated as an offset from the end of the QCSet. Note: If the provided index is negative, Then search QCSet from front to back. If the provided index is 0, the entire QCSet will be searched. Default value: 0 (search the entire QCSet).

Back: The first index of the element in the QCSet; -1 (if not found).
QCSet.prototype.has(arg), QCSet.prototype.contains(arg)
Find arg in the QCSet element, if it exists, return true, otherwise, return false.
QCSet.prototype.add([arg1[, argN])
Add one or more elements to the end of the QCSet data and return the new length value of the array.
QCSet.prototype.remove([arg1[, argN])
Remove one or more elements from the QCSet.
QCSet.prototype.each(callback)

Go through QCSet

callback: callback function.

Parameters

value: The current element being processed in the QCSet.

index [Optional]: The index of value in QCSet.

thisQCSet [Optional]: When the callback function is executed, it is used as the value of this.

Tips

return true skip the following code and continue the loop. Similar to continue.

return false to stop the loop and skip the following code. Similar to break.

QCSet.prototype.merge(arg)
Combine arg into the current QCSet and return the current QCSet.
QCSet.prototype.join(split)
Concatenate all the elements of QCSet into a string.
QCSet.prototype.equals(obj)
Compare whether two QCSet objects are the same, and return boolean.

Example

        function print(qcs) {
            console.log(qcs.join(","));
        }

        var qcs = QCSet("a, b, c, d");

        qcs.add(1, 2, 3, 4);
        print(qcs);             // a,b,c,d,1,2,3,4

        qcs.remove('b');
        print(qcs);             // a,c,d,1,2,3,4

        qcs.each(function (v, i) {
            console.log("'index': ", i, ",", "'value': ", v);
            if (i < 3)
                return true;
            else
                return false
            console.log("not print.");
        });
        // 'index':  0 , 'value':  a
        // 'index':  1 , 'value':  c
        // 'index':  2 , 'value':  d
        // 'index':  3 , 'value':  1

        qcs.indexOf('d');       // 2
        qcs.indexOf(3, -2);     // 5
        qcs.indexOf(3, -1);     // -1

        qcs.has(3);             // true
        qcs.has(8);             // false

        var qcs1 = QCSet([1,2,3]);
        var qcs2 = QCSet("a, b, c");

        qcs1.merge(qcs2);
        print(qcs1);            // 1,2,3,a,b,c

        print(qcs2);            // a,b,c

        qcs1.join(";");         // 1;2;3;a;b;c

        var ary = [1, 2, 3];
        qcs1 = QCSet(ary);
        qcs2 = QCSet([1, 2, 3]);
        qcs1.equals(qcs2);      // true

        var qcs3 = QCSet([1, 2, 4]);
        qcs1.equals(qcs3);      // false
