colorpicker

拾色器

静态调用

Code

        // html body
        <button id="color1" qc-for="colorpicker">
                Show colorpicker</button>

        <button id="color2" qc-for="colorpicker" qc-target="#text"
            qc-colorpicker-mode="list" qc-clolorpicker-css="background-color">
                Show colorpicker (background-color)</button>
        <span id="text"> text bgColor </span>

Result

text bgColor

动态调用

Code

        // html body
        <button id="color3">dyShow with panel</button>
        <button id="color4">dyShow with list</button>
        <span id="text2"> text bgColor </span>

        // javascript
        qc(function () {
            qc("#color3").click(function () {
                var obj = qc(this);
                qc.colorpicker.dyShow(obj, obj.css("color"), "panel", function(color) {
                    obj.css("color", color);
                });
            });

            qc("#color4").click(function () {
                var obj = qc("#text1");
                qc.colorpicker.dyShow(obj, obj.css("background-color"), "list", function(color) {
                    obj.css("background-color", color);
                });
            });

        });

Result

text bgColor
