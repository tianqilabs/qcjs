Extensions

Description: The following property methods of Array and String are expanded.

Array.prototype.contains(arg)
Find arg in the array element, if it exists, return true, otherwise, return false.

Code

        // javascript
        var ary = ["a", "b", "c", "d"];

        console.log(ary.contains("b"));     // true

        console.log(ary.contains("z"));     // false
Array.prototype.each(callback)
Traverse the array elements.
callback
Callback.

arr: The current element.

indx: current location.

arrs: Array object.

Code

        // javascript
        var ary = ["a", "b", "c", "d"];

        ary.each(function(arr, index, arrs) {
            console.log("value: ", arr, ",", "index: ", index, ",", "array: ", arrs);
        });

        // value:  a , index:  0 , array:  ["a", "b", "c", "d"]
        // value:  b , index:  1 , array:  ["a", "b", "c", "d"]
        // value:  c , index:  2 , array:  ["a", "b", "c", "d"]
        // value:  d , index:  3 , array:  ["a", "b", "c", "d"]
Array.prototype.equals(arg)
Array comparison. The same returns true, otherwise it returns false.

Code

        // javascript
        var ary1 = ["a", "b", "c", "d"];
        var ary2 = "a,b,c,d".split(",");

        var re = ary1.equals(ary2);
        console.log(re);                // true;
